function loadDoc() { const xhttp = new XMLHttpRequest(); xhttp.onreadystatechange = function() { if (this.readyState == 4 && this.status == 200) { var obj_arr = JSON.parse(this.responseText); document.getElementById("medicineId").src = obj_arr[0].id; document.getElementById("m_name").src = obj_arr[0].id; document.getElementById("man_date").src = obj_arr[0].id; document.getElementById("ex_date").src = obj_arr[0].id; document.getElementById("m_category").src = obj_arr[0].id; document.getElementById("stock").src = obj_arr[0].id; } }; xhttp.open("GET", "http://127.0.0.1:8000/api/medicine/list"); xhttp.send(); }
